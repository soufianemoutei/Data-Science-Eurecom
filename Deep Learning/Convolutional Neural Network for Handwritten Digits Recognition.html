Figure 1: Lenet-5

The LeNet architecture takes a 28x28xC image as input, where C is the number of color channels. Since MNIST images are grayscale, C is 1 in this case.

Layer 1 - Convolution (5x5): The output shape should be 28x28x6. Activation: ReLU. MaxPooling: The output shape should be 14x14x6.

Layer 2 - Convolution (5x5): The output shape should be 10x10x16. Activation: ReLU. MaxPooling: The output shape should be 5x5x16.

Flatten: Flatten the output shape of the final pooling layer such that it's 1D instead of 3D. You may need to use tf.reshape.

Layer 3 - Fully Connected: This should have 120 outputs. Activation: ReLU.

Layer 4 - Fully Connected: This should have 84 outputs. Activation: ReLU.

Layer 5 - Fully Connected: This should have 10 outputs. Activation: softmax.

# Functions for weigths and bias initilization 
def weight_variable(shape):
  initial = tf.truncated_normal(shape, stddev=0.1)
  return tf.Variable(initial)

def bias_variable(shape):
  initial = tf.constant(0., shape=shape)
  return tf.Variable(initial)

from tensorflow.contrib.layers import flatten

def LeNet5_Model(image):

    # Layer 1:
    # Convolution (5x5): The output shape should be 28x28x6. 
    conv1_weight = weight_variable([5,5,1,6])
    conv1_bias = bias_variable([6])
    conv1 = tf.nn.conv2d(image, conv1_weight, strides = [1,1,1,1], padding = 'SAME') + conv1_bias
    # Activation: ReLU.
    activation1 = tf.nn.relu(conv1)
    # MaxPooling: The output shape should be 14x14x6.
    layer1 = tf.nn.max_pool(activation1, ksize = [1,2,2,1], strides = [1,2,2,1], padding = 'VALID')


    # Layer 2:
    # Convolution (5x5): The output shape should be 10x10x16. 
    conv2_weight = weight_variable([5,5,6,16])
    conv2_bias = bias_variable([16])
    conv2 = tf.nn.conv2d(layer1, conv2_weight, strides = [1,1,1,1], padding = 'VALID') + conv2_bias
    # Activation: ReLU.
    activation2 = tf.nn.relu(conv2)
    # MaxPooling: The output shape should be 5x5x16.
    layer2 = tf.nn.max_pool(activation2, ksize = [1,2,2,1], strides = [1,2,2,1], padding = 'VALID')

    # Flatten: Flatten the output shape of the final pooling layer such that it's 1D instead of 3D. 
    flatten_v = flatten(layer2)

    # Layer 3:
    # Fully Connected: This should have 120 outputs.
    fc3_weight = weight_variable((400, 120))
    fc3_bias = bias_variable([120])
    fc3 = tf.matmul(flatten_v, fc3_weight) + fc3_bias
    # Activation: ReLU
    layer3 = tf.nn.relu(fc3)

    # Layer 4:
    # Fully Connected: This should have 84 outputs.
    fc4_weight = weight_variable((120, 84))
    fc4_bias = bias_variable([84])
    fc4 = tf.matmul(layer3, fc4_weight) + fc4_bias
    # Activation: ReLU
    layer4 = tf.nn.relu(fc4)

    # Layer 5:
    # Fully Connected: This should have 10 outputs
    fc5_weight = weight_variable((84, 10))
    fc5_bias = bias_variable([10])
    fc5 = tf.matmul(layer4, fc5_weight) + fc5_bias
    # Activation: softmax
    layer5 = tf.nn.softmax(fc5)

    return layer5

def LeNet5_Model_Dropout(image, keep_prob=1.):
    # Layer 1:
    # Convolution (5x5): The output shape should be 28x28x6. 
    conv1_weight = weight_variable([5,5,1,6])
    conv1_bias = bias_variable([6])
    conv1 = tf.nn.conv2d(image, conv1_weight, strides = [1,1,1,1], padding = 'SAME') + conv1_bias
    # Activation: ReLU.
    activation1 = tf.nn.relu(conv1)
    # MaxPooling: The output shape should be 14x14x6.
    layer1 = tf.nn.max_pool(activation1, ksize = [1,2,2,1], strides = [1,2,2,1], padding = 'VALID')


    # Layer 2:
    # Convolution (5x5): The output shape should be 10x10x16. 
    conv2_weight = weight_variable([5,5,6,16])
    conv2_bias = bias_variable([16])
    conv2 = tf.nn.conv2d(layer1, conv2_weight, strides = [1,1,1,1], padding = 'VALID') + conv2_bias
    # Activation: ReLU.
    activation2 = tf.nn.relu(conv2)
    # MaxPooling: The output shape should be 5x5x16.
    layer2 = tf.nn.max_pool(activation2, ksize = [1,2,2,1], strides = [1,2,2,1], padding = 'VALID')

    # Flatten: Flatten the output shape of the final pooling layer such that it's 1D instead of 3D. 
    flatten_v = flatten(layer2)

    # Try to add dropout (keep_prob) before the first fully connected layer. 
    flatten_v = tf.nn.dropout(x=flatten_v, keep_prob=keep_prob)

    # Layer 3:
    # Fully Connected: This should have 120 outputs.
    fc3_weight = weight_variable((400, 120))
    fc3_bias = bias_variable([120])
    fc3 = tf.matmul(flatten_v, fc3_weight) + fc3_bias
    # Activation: ReLU
    layer3 = tf.nn.relu(fc3)

    # Layer 4:
    # Fully Connected: This should have 84 outputs.
    fc4_weight = weight_variable((120, 84))
    fc4_bias = bias_variable([84])
    fc4 = tf.matmul(layer3, fc4_weight) + fc4_bias
    # Activation: ReLU
    layer4 = tf.nn.relu(fc4)

    # Layer 5:
    # Fully Connected: This should have 10 outputs
    fc5_weight = weight_variable((84, 10))
    fc5_bias = bias_variable([10])
    fc5 = tf.matmul(layer4, fc5_weight) + fc5_bias
    # Activation: softmax
    layer5 = tf.nn.softmax(fc5)

    return layer5
